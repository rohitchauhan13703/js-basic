JavaScript Basics

Foundation for Learning React

1. Variables (let, const)


// Using let (mutable)
let userName = "John";
userName = "Jane"; // Can be changed

// Using const (immutable)
const birthYear = 1990;
// birthYear = 2000; // Error: Can't reassign const

2. Data Types


// String
let greeting = "Hello World!";

// Number
let temperature = 23.5;

// Boolean
let isLoggedIn = true;

// Object
let user = { 
    name: "Alice", 
    age: 25 
};

// Array
let colors = ["red", "green", "blue"];

3. Functions


// Regular Function
function greet(name) {
    return `Hello, ${name}!`;
}

// Arrow Function
const addNumbers = (a, b) => a + b;

// Immediately Invoked Function Expression (IIFE)
(function() {
    console.log("This runs immediately");
})();

4. Conditional Statements


let score = 85;
let grade;

if (score >= 90) {
    grade = "A";
} else if (score >= 80) {
    grade = "B";
} else if (score >= 70) {
    grade = "C";
} else {
    grade = "D";
}

// Ternary operator
let status = score >= 50 ? "Pass" : "Fail";

5. Loops


// For loop
for (let i = 1; i <= 5; i++) {
    console.log(`Count: ${i}`);
}

// While loop
let x = 0;
while (x < 3) {
    console.log(`While: ${x}`);
    x++;
}

// Array iteration
const fruits = ["apple", "banana", "orange"];
fruits.forEach(fruit => console.log(fruit));

6. Arrays


const numbers = [1, 2, 3, 4, 5];

// Adding/removing elements
numbers.push(6); // end
numbers.pop(); // remove end
numbers.unshift(0); // beginning
numbers.shift(); // remove beginning

// Common array methods
const doubled = numbers.map(num => num * 2);
const evens = numbers.filter(num => num % 2 === 0);
const sum = numbers.reduce((total, num) => total + num, 0);

7. Objects


const person = {
    name: "John",
    age: 30,
    hobbies: ["sports", "music"],
    address: {
        street: "123 Main St",
        city: "Boston"
    },
    greet: function() {
        return `Hello, my name is ${this.name}`;
    }
};

// Accessing properties
console.log(person.name); // dot notation
console.log(person["age"]); // bracket notation
console.log(person.greet());

// Object destructuring
const { name, age } = person;

8. ES6 Features


// Template literals
const username = "Sarah";
console.log(`Hello, ${username}!`);

// Spread operator
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5];

// Rest parameters
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

// Classes
class Person {
    constructor(name) {
        this.name = name;
    }
    greet() {
        return `Hello, I'm ${this.name}`;
    }
}

9. Asynchronous JavaScript


// Callback
function fetchData(callback) {
    setTimeout(() => {
        callback("Data received");
    }, 1000);
}

// Promise
function getUser() {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve({ name: "John", id: 1 });
        }, 1500);
    });
}

// Async/Await
async function displayUser() {
    try {
        const user = await getUser();
        console.log(user);
    } catch (error) {
        console.log("Error:", error);
    }
}
